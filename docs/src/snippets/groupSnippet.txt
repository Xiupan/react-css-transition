import * as React from "react";
import { CSSTransitionGroup } from "react-css-transition";

import { FadeInOut } from "./transitions";

export const FadeInOutGroup = (props) => (
  <CSSTransitionGroup {...props}>
    {
      React.Children.map(
        props.children,
        (child) => <FadeInOut>{child}</FadeInOut>,
      )
    }
  </CSSTransitionGroup>
);
